## Algunas fórmulas que pueden ser útiles:

$$
\vec{a} \times \vec{b} = \begin{vmatrix} \vec{i} & \vec{j} & \vec{k} \\ a_{1} & a_{2} & a_{3} \\ b_{1} & b_{2} & b_{3} \end{vmatrix} = \begin{vmatrix} a_{2} & a_{3} \\ b_{2} & b_{3} \end{vmatrix} \vec{i} - \begin{vmatrix} a_{1} & a_{3} \\ b_{1} & b_{3} \end{vmatrix} \vec{j} + \begin{vmatrix} a_{1} & a_{2} \\ b_{1} & b_{2} \end{vmatrix} \vec{k} = (a_{2}b_{3} - a_{3}b_{2})\vec{i} - (a_{1}b_{3} - a_{3}b_{1})\vec{j} + (a_{1}b_{2} - a_{2}b_{1})\vec{k}
$$

$$
\text{Área del paralelogramo} = \|\vec{a} \times \vec{b}\| \qquad \text{Volumen del paralelepípedo: } V = \|(\vec{a}, \vec{b}, \vec{c})\|
$$

$$
\vec{U} \cdot \vec{V} = \|\vec{U}\| \|\vec{V}\| \cos \theta = u_{1}v_{1} + u_{2}v_{2} + u_{3}v_{3}
$$

$$
\text{Ecuaciones de la recta: } \begin{cases} x = x_{0} + ta \\ y = y_{0} + tb \\ z = z_{0} + tc \end{cases} \qquad t = \frac{x - x_{0}}{a} = \frac{y - y_{0}}{b} = \frac{z - z_{0}}{c}
$$
